<script>
    function calculate() {
        const exprStr = document.getElementById('func').value;
        const aStr = document.getElementById('lower').value;
        const bStr = document.getElementById('upper').value;
        const resultBox = document.getElementById('resultBox');

        try {
            // Evaluate bounds (allows inputs like 'pi' or '2*pi')
            const a = math.evaluate(aStr);
            const b = math.evaluate(bStr);

            // Compile the function for fast evaluation
            const compiled = math.compile(exprStr);
            const f = (x) => compiled.evaluate({ x: x });

            // Numerical Integration using Simpson's 1/3 Rule
            let n = 2000; // Even number of intervals
            let h = (b - a) / n;
            let sum = f(a) + f(b);
            
            for (let i = 1; i < n; i++) {
                let x = a + i * h;
                sum += f(x) * (i % 2 === 0 ? 2 : 4);
            }
            let result = (h / 3) * sum;

            // Show result
            resultBox.style.display = 'block';
            resultBox.innerHTML = `Area = <strong>${math.round(result, 6)}</strong>`;

            // Draw the graph
            drawGraph(f, a, b);

        } catch (err) {
            resultBox.style.display = 'block';
            resultBox.style.borderLeftColor = 'red';
            resultBox.style.backgroundColor = '#fde8e8';
            resultBox.innerHTML = `<span style="color: red;"><strong>Error:</strong> ${err.message}. Please check your math syntax.</span>`;
        }
    }

    function drawGraph(f, a, b) {
        // Determine the x-axis range to show a bit past the bounds
        const margin = Math.abs(b - a) * 0.3 || 2;
        const xMin = Math.min(a, b) - margin;
        const xMax = Math.max(a, b) + margin;
        
        // Generate coordinates for the function line
        const xLine = [];
        const yLine = [];
        for (let x = xMin; x <= xMax; x += (xMax - xMin) / 500) {
            xLine.push(x);
            yLine.push(f(x));
        }

        // Generate coordinates for the shaded integral area
        const xArea = [];
        const yArea = [];
        const start = Math.min(a, b);
        const end = Math.max(a, b);
        for (let x = start; x <= end; x += (end - start) / 200) {
            xArea.push(x);
            yArea.push(f(x));
        }

        // Plotly Trace for the outline of the function
        const traceLine = {
            x: xLine,
            y: yLine,
            type: 'scatter',
            mode: 'lines',
            name: 'f(x)',
            line: { color: '#2a75d3', width: 2 }
        };

        // Plotly Trace for the shaded area under the curve
        const traceArea = {
            x: xArea,
            y: yArea,
            fill: 'tozeroy',
            type: 'scatter',
            mode: 'none',
            name: `Integral`,
            fillcolor: 'rgba(42, 117, 211, 0.3)'
        };

        const layout = {
            title: 'Area Under Curve',
            margin: { t: 40, l: 40, r: 40, b: 40 },
            showlegend: false,
            xaxis: { zeroline: true, zerolinecolor: '#999' },
            yaxis: { zeroline: true, zerolinecolor: '#999' }
        };

        Plotly.newPlot('plot', [traceArea, traceLine], layout, { displayModeBar: false });
    }

    // Run once on load to show initial graph
    window.onload = calculate;
</script>
